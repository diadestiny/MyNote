/**
 * 实现单例模式：
 * 1.构造方法私有化
 * 2.对外提供一个公开的静态的获取当前对象的方法
 * 3.提供一个当前类型的静态变量
 */
//懒汉式单例模式
public class User {
	private static  User user=null;
	private User() {}
	public static User getInstance() {
		if(user==null) {
			user=new User();
		}
		return user;		
	}
	

}
/**
 * 饿汉式单例模式:
 * 在类加载阶段就创建了对象
 */
public class User {
	private static  User user=new User();
	private User() {}
	public static User getInstance() {		
		return user;		
	}
}
